#include int main(){ int s; int t; scanf("%d %d",&s,&t); int a; int b; scanf("%d %d",&a,&b); int m; int n; int ans1=0,ans2=0; scanf("%d %d",&m,&n); int *apple = malloc(sizeof(int) * m); for(i = 0; i < m; i++){ scanf("%d",&apple[i]); if((apple[i]+a)<=t && (apple[i]+a)>=s) ans1++; } int *orange = malloc(sizeof(int) * n); for(int i = 0; i < n; i++){ scanf("%d",&orange[orange_i]); if((orange[i]+b)<=t && (orange[i]+b)>=s) ans2++; } printf("%d\n%d",ans1,ans2); return 0; }
